403def integer_squareroot(n: uint64) -> uint64: 

404 """ 

405 Return the largest integer ``x`` such that ``x**2 <= n``. 

406 """ 

407 x = n 

408 y = (x + 1) // 2 

409 while y < x: 

410 x = y

505def compute_shuffled_index(index: uint64, index_count: uint64, seed: Bytes32) -> uint64: 

506 """ 

507 Return the shuffled index corresponding to ``seed`` (and ``index_count``). 

508 """ 

509 assert index < index_count 

510 

511 # Swap or not (https://link.springer.com/content/pdf/10.1007%2F978-3-642-32009-5_1.pdf) 

512 # See the 'generalized domain' algorithm on page 3 

513 for current_round in range(SHUFFLE_ROUND_COUNT): 

514 pivot = bytes_to_uint64(hash(seed + uint_to_bytes(uint8(current_round)))[0:8]) % index_count 

515 flip = (pivot + index_count - index) % index_count 

516 position = max(index, flip) 

517 source = hash( 

518 seed 

519 + uint_to_bytes(uint8(current_round)) 

520 + uint_to_bytes(uint32(position // 256)) 

521 ) 

522 byte = uint8(source[(position % 256) // 8]) 

523 bit = (byte >> (position % 8)) % 2 

524 index = flip if bit else index 

525 

526 return index

529def compute_proposer_index(state: BeaconState, indices: Sequence[ValidatorIndex], seed: Bytes32) -> ValidatorIndex: 

530 """ 

531 Return from ``indices`` a random index sampled by effective balance. 

532 """ 

533 assert len(indices) > 0 

534 MAX_RANDOM_BYTE = 2**8 - 1 

535 i = uint64(0) 

536 total = uint64(len(indices)) 

537 while True: 

538 candidate_index = indices[compute_shuffled_index(i % total, total, seed)] 

539 random_byte = hash(seed + uint_to_bytes(uint64(i // 32)))[i % 32] 

540 effective_balance = state.validators[candidate_index].effective_balance 

541 if effective_balance * MAX_RANDOM_BYTE >= MAX_EFFECTIVE_BALANCE * random_byte: 

542 return candidate_index 

543 i += 1

778def initiate_validator_exit(state: BeaconState, index: ValidatorIndex) -> None: 

779 """ 

780 Initiate the exit of the validator with index ``index``. 

781 """ 

782 # Return if validator already initiated exit 

783 validator = state.validators[index] 

784 if validator.exit_epoch != FAR_FUTURE_EPOCH: 

785 return 

786 

787 # Compute exit queue epoch 

788 exit_epochs = [v.exit_epoch for v in state.validators if v.exit_epoch != FAR_FUTURE_EPOCH] 

789 exit_queue_epoch = max(exit_epochs + [compute_activation_exit_epoch(get_current_epoch(state))]) 

790 exit_queue_churn = len([v for v in state.validators if v.exit_epoch == exit_queue_epoch]) 

791 if exit_queue_churn >= get_validator_churn_limit(state): 

792 exit_queue_epoch += Epoch(1) 

793 

794 # Set validator exit epoch and withdrawable epoch 

795 validator.exit_epoch = exit_queue_epoch 

796 validator.withdrawable_epoch = Epoch(validator.exit_epoch + config.MIN_VALIDATOR_WITHDRAWABILITY_DELAY)

799def slash_validator(state: BeaconState, 

800 slashed_index: ValidatorIndex, 

801 whistleblower_index: ValidatorIndex=None) -> None: 

802 """ 

803 Slash the validator with index ``slashed_index``. 

804 """ 

805 epoch = get_current_epoch(state) 

806 initiate_validator_exit(state, slashed_index) 

807 validator = state.validators[slashed_index] 

808 validator.slashed = True 

809 validator.withdrawable_epoch = max(validator.withdrawable_epoch, Epoch(epoch + EPOCHS_PER_SLASHINGS_VECTOR))

815 if whistleblower_index is None: 815 ↛ 817line 815 didn't jump to line 817, because the condition on line 815 was never false

816 whistleblower_index = proposer_index 

817 whistleblower_reward = Gwei(validator.effective_balance // WHISTLEBLOWER_REWARD_QUOTIENT) 

818 proposer_reward = Gwei(whistleblower_reward // PROPOSER_REWARD_QUOTIENT)

958def process_justification_and_finalization(state: BeaconState) -> None: 

959 # Initial FFG checkpoint values have a `0x00` stub for `root`. 

960 # Skip FFG updates in the first two epochs to avoid corner cases that might result in modifying this stub.

971def weigh_justification_and_finalization(state: BeaconState, 

972 total_active_balance: Gwei, 

973 previous_epoch_target_balance: Gwei, 

974 current_epoch_target_balance: Gwei) -> None: 

975 previous_epoch = get_previous_epoch(state) 

976 current_epoch = get_current_epoch(state) 

977 old_previous_justified_checkpoint = state.previous_justified_checkpoint 

978 old_current_justified_checkpoint = state.current_justified_checkpoint 

979 

980 # Process justifications 

981 state.previous_justified_checkpoint = state.current_justified_checkpoint 

982 state.justification_bits[1:] = state.justification_bits[:JUSTIFICATION_BITS_LENGTH - 1] 

983 state.justification_bits[0] = 0b0 

984 if previous_epoch_target_balance * 3 >= total_active_balance * 2: 

985 state.current_justified_checkpoint = Checkpoint(epoch=previous_epoch, 

986 root=get_block_root(state, previous_epoch)) 

987 state.justification_bits[1] = 0b1 

988 if current_epoch_target_balance * 3 >= total_active_balance * 2: 

989 state.current_justified_checkpoint = Checkpoint(epoch=current_epoch, 

990 root=get_block_root(state, current_epoch)) 

991 state.justification_bits[0] = 0b1 

992 

993 # Process finalizations 

994 bits = state.justification_bits 

995 # The 2nd/3rd/4th most recent epochs are justified, the 2nd using the 4th as source 

996 if all(bits[1:4]) and old_previous_justified_checkpoint.epoch + 3 == current_epoch: 

997 state.finalized_checkpoint = old_previous_justified_checkpoint 

998 # The 2nd/3rd most recent epochs are justified, the 2nd using the 3rd as source 

999 if all(bits[1:3]) and old_previous_justified_checkpoint.epoch + 2 == current_epoch: 

1000 state.finalized_checkpoint = old_previous_justified_checkpoint 

1001 # The 1st/2nd/3rd most recent epochs are justified, the 1st using the 3rd as source 

1002 if all(bits[0:3]) and old_current_justified_checkpoint.epoch + 2 == current_epoch: 

1003 state.finalized_checkpoint = old_current_justified_checkpoint 

1004 # The 1st/2nd most recent epochs are justified, the 1st using the 2nd as source 

1005 if all(bits[0:2]) and old_current_justified_checkpoint.epoch + 1 == current_epoch: 

1006 state.finalized_checkpoint = old_current_justified_checkpoint

1035def get_attestation_component_deltas(state: BeaconState, 

1036 attestations: Sequence[PendingAttestation] 

1037 ) -> Tuple[Sequence[Gwei], Sequence[Gwei]]: 

1038 """ 

1039 Helper with shared logic for use by get source, target, and head deltas functions 

1040 """ 

1041 rewards = [Gwei(0)] * len(state.validators) 

1042 penalties = [Gwei(0)] * len(state.validators) 

1043 total_balance = get_total_active_balance(state) 

1044 unslashed_attesting_indices = get_unslashed_attesting_indices(state, attestations) 

1045 attesting_balance = get_total_balance(state, unslashed_attesting_indices) 

1046 for index in get_eligible_validator_indices(state): 

1047 if index in unslashed_attesting_indices: 

1048 increment = EFFECTIVE_BALANCE_INCREMENT # Factored out from balance totals to avoid uint64 overflow 

1049 if is_in_inactivity_leak(state): 

1050 # Since full base reward will be canceled out by inactivity penalty deltas, 

1051 # optimal participation receives full base reward compensation here. 

1052 rewards[index] += get_base_reward(state, index) 

1053 else: 

1054 reward_numerator = get_base_reward(state, index) * (attesting_balance // increment) 

1055 rewards[index] += reward_numerator // (total_balance // increment) 

1056 else: 

1057 penalties[index] += get_base_reward(state, index) 

1058 return rewards, penalties

1105def get_inactivity_penalty_deltas(state: BeaconState) -> Tuple[Sequence[Gwei], Sequence[Gwei]]: 

1106 """ 

1107 Return inactivity reward/penalty deltas for each validator. 

1108 """ 

1109 penalties = [Gwei(0) for _ in range(len(state.validators))] 

1110 if is_in_inactivity_leak(state): 

1111 matching_target_attestations = get_matching_target_attestations(state, get_previous_epoch(state)) 

1112 matching_target_attesting_indices = get_unslashed_attesting_indices(state, matching_target_attestations) 

1113 for index in get_eligible_validator_indices(state): 

1114 # If validator is performing optimally this cancels all rewards for a neutral balance 

1115 base_reward = get_base_reward(state, index) 

1116 penalties[index] += Gwei(BASE_REWARDS_PER_EPOCH * base_reward - get_proposer_reward(state, index)) 

1117 if index not in matching_target_attesting_indices: 

1118 effective_balance = state.validators[index].effective_balance 

1119 penalties[index] += Gwei(effective_balance * get_finality_delay(state) // INACTIVITY_PENALTY_QUOTIENT)

1852def compute_weak_subjectivity_period(state: BeaconState) -> uint64: 

1853 """ 

1854 Returns the weak subjectivity period for the current ``state``. 

1855 This computation takes into account the effect of: 

1856 - validator set churn (bounded by ``get_validator_churn_limit()`` per epoch), and 

1857 - validator balance top-ups (bounded by ``MAX_DEPOSITS * SLOTS_PER_EPOCH`` per epoch). 

1858 A detailed calculation can be found at: 

1859 https://github.com/runtimeverification/beacon-chain-verification/blob/master/weak-subjectivity/weak-subjectivity-analysis.pdf 

1860 """ 

1861 ws_period = config.MIN_VALIDATOR_WITHDRAWABILITY_DELAY 

1862 N = len(get_active_validator_indices(state, get_current_epoch(state))) 

1863 t = get_total_active_balance(state) // N // ETH_TO_GWEI 

1864 T = MAX_EFFECTIVE_BALANCE // ETH_TO_GWEI 

1865 delta = get_validator_churn_limit(state) 

1866 Delta = MAX_DEPOSITS * SLOTS_PER_EPOCH 

1867 D = SAFETY_DECAY 

1868 

1869 if T * (200 + 3 * D) < t * (200 + 12 * D): 

1870 epochs_for_validator_set_churn = ( 

1871 N * (t * (200 + 12 * D) - T * (200 + 3 * D)) // (600 * delta * (2 * t + T)) 

1872 ) 

1873 epochs_for_balance_top_ups = ( 

1874 N * (200 + 3 * D) // (600 * Delta) 

1875 ) 

1876 ws_period += max(epochs_for_validator_set_churn, epochs_for_balance_top_ups) 

1877 else: 

1878 ws_period += ( 

1879 3 * N * D * t // (200 * Delta * (T - t)) 

1880 ) 

1881 

1882 return ws_period
